CMAKE_MINIMUM_REQUIRED(VERSION 3.8.0)

INCLUDE_DIRECTORIES(CodeGen)

TARGET_LINK_LIBRARIES(Trinity.TSL.CodeGen Trinity.TSL.Parser)

    INSTALL(FILES ${TRINITY_LIB_DIR}/Trinity.TSL.CodeGen DESTINATION ${TRINITY_OUTPUT_DIR})

    INSTALL(FILES ${TRINITY_LIB_DIR}/Trinity.TSL.CodeGen.exe DESTINATION ${TRINITY_OUTPUT_DIR})
ENDIF()
